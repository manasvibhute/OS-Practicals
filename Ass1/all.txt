file="address.txt"

# Create file if it doesnâ€™t exist
if [ ! -f "$file" ]; then
    touch "$file"
    echo "ID;Name;Phone" > "$file"
fi

generate_id() {
    if [ "$(wc -l < "$file")" -le 1 ]; then
        echo 1
    else
        tail -n +2 "$file" | cut -d';' -f1 | sort -n | tail -1 | awk '{print $1+1}'
    fi

add_entry() {

    echo "Entry added!"

    read -p "Enter Name or Phone to search: " key
    grep -i "$key" "$file" || echo "No match found."
}

    if grep -q "^$id;" "$file"; then
        grep -v "^$id;" "$file" > temp && mv temp "$file"
        echo "Entry deleted!"
    else
        echo "No record found."
    fi
}

while true; do
    echo ""
    echo "==== Address Book ===="
    echo "1. Search Address Book"
    echo "2. Add Entry"
    echo "3. Remove Entry"
    echo "4. Quit"
    read -p "Choice: " ch
    case $ch in
        1) search_entry ;;
        2) add_entry ;;
        3) remove_entry ;;
        4) echo "Goodbye!"; exit ;;
        *) echo "Invalid choice" ;;
    esac

#!/bin/bash

    echo -e "First\tLast\tPhone" > "$file"

add_entry() {
    read -p "Enter First Name: " first

    read -p "Enter Name or Phone to search: " key
    grep -i "$key" "$file" || echo "No match found."

sort_phonebook() {

    tail -n +2 "$file" | sort -k2 >> temp
    mv temp "$file"

}

delete_entry() {
    read -p "Enter phone or name to delete: " key
    grep -iv "$key" "$file" > temp && mv temp "$file"
    echo "Deleted."
}

    echo "2. Search Entry"
    echo "3. Sort by Last Name"
    echo "4. Delete Entry"
    echo "5. Exit"
    read -p "Choice: " ch
    case $ch in
        1) add_entry ;;
        2) search_entry ;;
        3) sort_phonebook ;;
        4) delete_entry ;;
        5) exit ;;

#!/bin/bash

test_file() {
    read -p "Enter filename: " fname
    if [ -f "$fname" ]; then
        echo "File exists."
    else
        echo "File not found."
    fi
}

read_file() {
    read -p "Enter filename: " fname
    if [ -f "$fname" ]; then
        cat "$fname"
    else
        echo "File not found."
    fi
}

delete_file() {
    read -p "Enter filename: " fname
    if [ -f "$fname" ]; then
        rm "$fname"
        echo "File deleted."
    else
        echo "File not found."
    fi
}

list_files() {
    echo "Files in current directory:"
    ls -l
}

while true; do
    echo ""
    echo "==== File Manager ===="
    echo "1. Test if file exists"
    echo "2. Read a file"
    echo "3. Delete a file"
    echo "4. List files"
    echo "5. Exit"
    read -p "Choice: " ch
    case $ch in
        1) test_file ;;
        2) read_file ;;
        3) delete_file ;;
        4) list_files ;;
        5) exit ;;
        *) echo "Invalid choice" ;;
    esac
done
